func (h *TransactionHandler) ExportPortfolioCSV(c echo.Context) error {

	userID := getUserID(c)

	// Get portfolio data
	data, err := h.Service.GetPortfolio(userID)
	if err != nil {
		return c.JSON(http.StatusInternalServerError, map[string]string{
			"error": err.Error(),
		})
	}

	// Set headers for file download
	c.Response().Header().Set("Content-Type", "text/csv")
	c.Response().Header().Set("Content-Disposition", "attachment;filename=portfolio.csv")

	writer := csv.NewWriter(c.Response())
	defer writer.Flush()

	// Write header
	header := []string{
		"Symbol",
		"Quantity",
		"Avg Buy Price",
		"Current Price",
		"Total Investment",
		"Current Value",
		"Profit/Loss",
	}
	writer.Write(header)

	// Write data rows
	for _, r := range data {
		record := []string{
			r.Symbol,
			fmt.Sprintf("%.2f", r.Quantity),
			fmt.Sprintf("%.2f", r.AvgBuyPrice),
			fmt.Sprintf("%.2f", r.CurrentPrice),
			fmt.Sprintf("%.2f", r.TotalInvestment),
			fmt.Sprintf("%.2f", r.CurrentValue),
			fmt.Sprintf("%.2f", r.ProfitLoss),
		}
		writer.Write(record)
	}

	return nil
}
