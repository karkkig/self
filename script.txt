package main

import (
	"context"
	"fmt"
	"net/url"
	"os"
	"time"

	"github.com/chromedp/chromedp"
	"github.com/chromedp/cdproto/page"
)

// Read HTML file
func readHTML(path string) (string, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return "", err
	}
	return string(data), nil
}

// Generate PDF using chromedp
func generatePDF(html string, output string) error {

	// Create context
	ctx, cancel := chromedp.NewContext(context.Background())
	defer cancel()

	// Add timeout
	ctx, cancel = context.WithTimeout(ctx, 20*time.Second)
	defer cancel()

	var pdfBuf []byte

	// Convert HTML to data URL (MOST RELIABLE METHOD)
	htmlURL := "data:text/html," + url.PathEscape(html)

	// Run tasks
	err := chromedp.Run(ctx,
		chromedp.Navigate(htmlURL),
		chromedp.Sleep(2*time.Second), // wait for rendering

		chromedp.ActionFunc(func(ctx context.Context) error {
			buf, _, err := page.PrintToPDF().
				WithPrintBackground(true).
				WithPaperWidth(8.27).  // A4 width
				WithPaperHeight(11.69). // A4 height
				Do(ctx)

			pdfBuf = buf
			return err
		}),
	)

	if err != nil {
		return err
	}

	// Save PDF
	return os.WriteFile(output, pdfBuf, 0644)
}

func main() {

	// Create output folder (optional)
	os.MkdirAll("output", os.ModePerm)

	// Invoice
	invoiceHTML, err := readHTML("invoice.html")
	if err != nil {
		panic(err)
	}
	err = generatePDF(invoiceHTML, "output/invoice.pdf")
	if err != nil {
		panic(err)
	}

	// QR
	qrHTML, err := readHTML("qr.html")
	if err != nil {
		panic(err)
	}
	err = generatePDF(qrHTML, "output/qr.pdf")
	if err != nil {
		panic(err)
	}

	// Agreement
	agreementHTML, err := readHTML("agreement.html")
	if err != nil {
		panic(err)
	}
	err = generatePDF(agreementHTML, "output/agreement.pdf")
	if err != nil {
		panic(err)
	}

	fmt.Println("✅ All PDFs generated successfully!")
}
