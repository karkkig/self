let stockData = [];

function loadPortfolio() {

    const token = localStorage.getItem("token");

    fetch(API + "/api/portfolio", {
        headers: { "Authorization": "Bearer " + token }
    })
    .then(res => res.json())
    .then(response => {

        stockData = response.data.stocks || [];   // 🔥 IMPORTANT

        renderTable();
    });
}
