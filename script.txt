package pdf

import (
	"github.com/johnfercher/maroto/v2/pkg/core"
)

// Generate a very basic PDF
func GenerateInvoice(path string) error {

	// Create empty Maroto instance (core only)
	m := core.NewMaroto(nil)

	// Add simple rows (no styling, no helpers)
	m.AddRow(20)
	m.AddRow(10)
	m.AddRow(10)

	// Generate document
	doc, err := m.Generate()
	if err != nil {
		return err
	}

	return doc.Save(path)
}
